procedure load_reserved_words is
begin
  for c_rec in ( select keyword from v$reserved_words
                 where (reserved = 'Y' or res_semi = 'Y')
                   and length(trim(keyword)) > 1 order by 1 ) loop
    g_reserved_words(c_rec.keyword) := 'Y';
  end loop;
end;


function is_reserved_word(i_word varchar2) return varchar2 is
is_reserved varchar2(1) := 'N';
begin
  begin
    is_reserved := g_reserved_words(upper(i_word));
  exception when others then
    is_reserved := 'N';
  end;
  return is_reserved;
end;



function valid_oracle_name(i_name varchar2) return varchar2 is
l_name varchar2(100) := upper(trim(i_name));
c_char varchar2(1);
n_pos integer;
begin
  if l_name is not null then
    
    l_name := regexp_replace(l_name,'[^A-Z,0-9,_]');
    
    c_char := substr(l_name, 1, 1);
    n_pos := instr('0123456789', c_char);
    if n_pos > 0 then
      l_name := 'N'||l_name;
    end if;
    
    c_char := substr(l_name, 1, 1);
    while c_char = '_' loop
      l_name := substr(l_name, 2);
      c_char := substr(l_name, 1, 1);
    end loop;
    
    if l_name is null then
      l_name := 'UNDERSCORE';
    end if;
    
    if xml2table.is_reserved_word(l_name) = 'Y' then
      l_name := l_name||'_';
    end if;
    
    l_name := substr(l_name, 1, 30);
  end if;
  return l_name;
end;

function interval_in_seconds(i_ts1 timestamp, i_ts2 timestamp) return number is

num_val number(19,6);
diff interval day(9) to second;
begin
  diff := (i_ts2 - i_ts1);
  num_val := extract(second from diff);
  num_val := num_val + extract(minute from diff) * 60;
  num_val := num_val + extract(hour from diff) * 60 * 60;
  num_val := num_val + extract(day from diff) * 60 * 60 * 24;
  return num_val;
end;


function find_dup_columns(p_colnames colnames) return varchar2 is
type tcols is table of number index by varchar2(30);
cols tcols;
l_current varchar2(30);
l_duplicates varchar2(4000) := '';
l_sep varchar2(1) := '';
begin
  for i in 1..p_colnames.count loop
    if cols.exists(p_colnames(i)) then
      cols(p_colnames(i)) := cols(p_colnames(i)) + 1;
    else
      cols(p_colnames(i)) := 1;
    end if;
  end loop;
  l_current := cols.first;
  while l_current is not null loop
    
    if cols(l_current) > 1 then
      l_duplicates := l_duplicates||l_sep||l_current;
      l_sep := ',';
    end if;
    l_current := cols.next(l_current);
  end loop;
  
  return chr(10)||'Duplicate columns:'||chr(10)||l_duplicates;
end;

procedure print_configuration is
l_current varchar2(30);

    ||' SAVEMESSAGE='||g_configurations(l_current).savemessage

procedure add_to_parts(i_path varchar2, i_denormno integer, i_tag varchar2, i_colname varchar2, i_value varchar2, i_instruction varchar2) is
i integer;
begin
  g_parts.extend(1);
  i := g_parts.count;
  g_parts(i).path := i_path;
  g_parts(i).denormno := i_denormno;
  g_parts(i).tag := i_tag;
  g_parts(i).colname := i_colname;
  g_parts(i).value := i_value;
  g_parts(i).instruction := i_instruction;
end;


procedure append_csv_to_parts(i_value varchar2) is
i integer;
eqpos integer;
tag varchar2(100) := '';

l_value varchar2(4000) := trim(replace(i_value, ',', ';'));
l_separator varchar2(5) := ',';
begin
  if l_value is not null then
    i := g_parts.count;
    eqpos := instr(i_value,'=');
    if eqpos > 1 then
      tag := substr(i_value,1,eqpos);
      if tag = substr(g_parts(i).value,1,eqpos) then
        
        l_separator := ','||chr(10);
      end if;
      
      while g_parts(i).instruction = 'DELETE' and i > 1 loop
        i := i - 1;
      end loop;
    end if;
    if g_parts(i).value is null then
      
      g_parts(i).value := l_value;
    else
      
      g_parts(i).value := g_parts(i).value||l_separator||l_value;
    end if;
  end if;
end;


procedure elaborate_error_message is
l_msg varchar2(4000);
begin
  
  if g_tracetag is not null then
    l_msg := g_message_store_rec.errormessage;
    l_msg := nvl(l_msg, 'OK.');
    l_msg := l_msg||chr(10)||to_char(g_parts.count - g_denormalize_count)||' parts. ';
    if g_denormalize_count > 0 then
      l_msg := l_msg||'Denormalized to '||to_char(g_denormalize_count)||' rows.';
    end if;
  else
    l_msg := null;
  end if;
  g_message_store_rec.errormessage := l_msg;
end;

function prefix(i_prefix varchar2, i_name varchar2) return varchar2 is
l_new_name varchar2(60) := trim(i_name);
l_prefix varchar2(30) := trim(i_prefix);
begin
  if l_new_name is not null then
    if l_prefix is not null then
      if substr(l_new_name,1,2) = 'L_' then
        l_new_name := 'L_'||l_prefix||substr(l_new_name,3);
      else
        l_new_name := l_prefix||l_new_name;
      end if;
    end if;
  end if;
  
  return l_new_name;
end;


function column_name(i_tagname varchar2, i_newcolumnname varchar2, i_prefix varchar2, i_isattribute boolean) return varchar2 is
l_new_name varchar2(60) := '?';
l_colon_pos integer := 0;
begin
  if i_newcolumnname is not null then
    l_new_name := i_newcolumnname;
  else
    if i_isattribute then
      
      
      if (upper(i_tagname) = 'XMLNS') or (instr(i_tagname, ':') > 1) then
        l_new_name := null;
      else
        l_new_name := i_tagname;
      end if;
    else
      
      
      l_colon_pos := instr(i_tagname,':');
      if l_colon_pos > 0 then
        l_new_name := substr(i_tagname, l_colon_pos + 1);
      else
        l_new_name := i_tagname;
      end if;
    end if;
  end if;
  l_new_name := xml2table.prefix(i_prefix, l_new_name);
  l_new_name := xml2table.valid_oracle_name(l_new_name);
  return l_new_name;
end;


procedure set_event_timestamp(i_eventtime varchar2, i_denormno number, i_dateformat varchar2, i_conversion varchar2, i_use_dbtimezone varchar2) is
l_dateformat varchar2(50);
l_eventtime varchar2(50) := i_eventtime;
begin
  
  
  
  
  if i_conversion = '3' then
    g_event_timestamp := g_message_store_rec.insertts;
  else
    
    begin
      l_dateformat :=  nvl(i_dateformat,'YYYY-MM-DD"T"HH24:MI:SS.FF"Z"');
      g_event_timestamp := to_timestamp_tz(i_eventtime, l_dateformat);
    exception when others then
      begin
        
        l_dateformat :=  nvl(i_dateformat, 'YYYY-MM-DD"T"HH24:MI:SS.FFTZH:TZM');
        g_event_timestamp := sys_extract_utc(to_timestamp_tz(i_eventtime, l_dateformat));
      exception when others then
        begin
          
          l_dateformat := 'YYYY-MM-DD HH24:MI:SS.FF';
          l_eventtime := replace(l_eventtime, 'T', ' ');
          l_eventtime := replace(l_eventtime, 'Z', '');
          g_event_timestamp := to_timestamp(l_eventtime, l_dateformat);
        exception when others then
          if i_conversion = '2' then
            g_message_store_rec.errormessage := substr('EVENTTIME: '||l_eventtime||' is incorrect: '
              ||sqlerrm||chr(10)||dbms_utility.format_error_backtrace,1,4000);
            g_message_store_rec.returncode := -6;
            xml2table.log_message;
          end if;
          g_event_timestamp := g_message_store_rec.insertts;
        end;
      end;
    end;
  end if;
  if i_use_dbtimezone = 'Y' then
    g_event_timestamp := from_tz(g_event_timestamp,'UTC') at time zone dbtimezone;
  end if;
  for i in reverse 1..i_denormno loop
    if not g_event_timestamps.exists(i) then
      g_event_timestamps(i) := g_event_timestamp;
    else
      exit;
    end if;
  end loop;
end;




procedure extract_node(i_domnode dbms_xmldom.domnode, i_path varchar2, i_prefix varchar2, i_tagdelimit boolean, i_denormno integer) is

l_nodename varchar2(100);
l_nodevalue varchar2(32767);
l_childnode dbms_xmldom.domnode;
l_childnodes dbms_xmldom.domnodelist;
l_nodelength integer;
l_nodemap dbms_xmldom.domnamednodemap;
l_attrnode dbms_xmldom.domnode;
l_attributecount integer;
l_attrname varchar2(100);
l_colname varchar2(30);
l_funname varchar2(30);
l_renamedcolumn varchar2(30);
l_attrvalue varchar2(32767);
l_path varchar2(32767);
l_prefix varchar2(30) := i_prefix;
l_tagdelimit boolean := i_tagdelimit;
l_denormno integer := i_denormno;
l_previous_path varchar2(32767);
l_instruction instruction_type;
l_attrinstruction instruction_type;
l_clob clob;
l_element_name varchar2(30);
l_element_node dbms_xmldom.domnode;
l_node dbms_xmldom.domnode;
begin
   
   l_nodename := dbms_xmldom.getnodename(i_domnode);

   
   l_path := i_path||'/'||l_nodename;

   
   
   if  g_parts.count > 0
   and g_parts(g_parts.count).instruction = 'FIRSTCHILDNAME'
   and (   g_parts(g_parts.count).value is null
        or substr(g_parts(g_parts.count).value,1,1)='#'
       )
   then
      
      
      if i_path = g_parts(g_parts.count).path
      then
         g_parts(g_parts.count).value := l_nodename;
      end if;
   end if;

   begin
     l_instruction := g_path_instructions(l_path);
   exception when others then
     l_instruction.instruction := '-';
     l_instruction.newname := null;
   end;

   
   l_colname := null;
   l_funname := null;
   if l_instruction.instruction = 'DENORMALIZE' then
     
     g_denormalize_count := g_denormalize_count + 1;
     l_denormno := g_denormalize_count;
     if l_instruction.newname is not null then
      add_to_parts(l_path, l_denormno, upper(l_instruction.newname), l_nodename, null, l_instruction.instruction);
     end if;
     add_to_parts(l_path, l_denormno, null, null, null, l_instruction.instruction);
   elsif l_instruction.instruction = 'RENAME' then
     l_colname := l_instruction.newname;
   elsif l_instruction.instruction = 'FIRSTCHILDNAME' then
     add_to_parts(l_path, l_denormno, upper(l_instruction.newname), l_instruction.newname, null, l_instruction.instruction);
   elsif l_instruction.instruction = 'SPLIT' then
     add_to_parts(l_path, l_denormno, null, null, null, l_instruction.instruction);
   elsif l_instruction.instruction = 'TAGDELIMIT' then
     add_to_parts(l_path, l_denormno, upper(l_instruction.newname), prefix(l_prefix, l_instruction.newname), null, l_instruction.instruction);
     l_tagdelimit := true;
   elsif l_instruction.instruction = 'PREFIX' then
     add_to_parts(l_path, l_denormno, upper(l_instruction.newname), null, null, l_instruction.instruction);
     l_prefix := l_instruction.newname;
   elsif l_instruction.instruction = 'DELETE' then
     add_to_parts(l_path, l_denormno, null, null, null, l_instruction.instruction);
   elsif nvl(l_instruction.instruction, '-') <> '-' then
     l_colname := l_instruction.newname;
     l_funname := l_instruction.instruction;
   end if;

    
   l_childnodes := dbms_xmldom.getchildnodes(i_domnode);

   if not dbms_xmldom.isnull(l_childnodes) then
     l_nodelength := dbms_xmldom.getlength(l_childnodes);

     
     

     
     l_nodemap := dbms_xmldom.getattributes(i_domnode);
     l_attributecount := dbms_xmldom.getlength(l_nodemap);

     

     
     for k in 0..l_attributecount-1
     loop
        l_attrnode := dbms_xmldom.item(l_nodemap, k);
        l_attrname := dbms_xmldom.getnodename(l_attrnode);
        
        if instr(l_attrname,':') = 0
        then
           begin
             l_attrinstruction := g_path_instructions(l_path||'/'||l_attrname);
           exception when others then
             l_attrinstruction.instruction := '-';
             l_attrinstruction.newname := null;
           end;
           
           if not l_attrinstruction.instruction = 'DELETE' then
             if l_attrinstruction.instruction = 'RENAME' then
               l_colname := l_attrinstruction.newname;
             end if;
             l_attrvalue := dbms_xmldom.getnodevalue(l_attrnode);
             add_to_parts(l_path||'/'||l_attrname, l_denormno, upper(l_attrname)
                         , xml2table.column_name(l_attrname, l_colname, null, true)
                         , l_attrvalue, l_attrinstruction.instruction
                         );
           end if;
        end if;
     end loop;

     
     if l_instruction.instruction = 'SPLIT' then
       g_split_nodes.extend;
       g_split_nodes(g_split_nodes.count) := dbms_xmldom.item(l_childnodes, l_nodelength-1);
       if dbms_xmldom.getnodetype(g_split_nodes(g_split_nodes.count)) = dbms_xmldom.text_node
       then
          
          
          
          l_element_name := l_instruction.newname;
          for i in reverse 1..g_parts.count - 1
          loop
             if l_instruction.newname is null
                or upper(g_parts(i).tag) = l_instruction.newname
             then
                l_element_name := g_parts(i).value;
                exit;
             end if;
          end loop;
          l_element_node := dbms_xmldom.makenode(dbms_xmldom.createelement(g_domdoc, l_element_name));
          l_node := dbms_xmldom.appendchild(l_element_node, dbms_xmldom.item(l_childnodes, l_nodelength-1));
          g_split_nodes(g_split_nodes.count) := l_element_node;
       elsif l_instruction.newname like '"%"'
       then
          l_element_node := dbms_xmldom.makenode(dbms_xmldom.createelement(g_domdoc, trim('"' from l_instruction.newname)));
          for i in 0..l_nodelength-1
          loop
             l_node := dbms_xmldom.appendchild(l_element_node, dbms_xmldom.item(l_childnodes, i));
          end loop;
          g_split_nodes(g_split_nodes.count) := l_element_node;
       end if;
     elsif not l_instruction.instruction = 'DELETE' then
       for i in 0..l_nodelength-1 loop
         l_childnode := dbms_xmldom.item(l_childnodes, i);
         
         if not dbms_xmldom.isnull(n => l_childnode) then
           if dbms_xmldom.getnodetype(l_childnode ) = dbms_xmldom.text_node then
             l_nodevalue := l_nodevalue||dbms_xmldom.getnodevalue(l_childnode);
           else
             extract_node(l_childnode, l_path, l_prefix, l_tagdelimit, l_denormno);
           end if;
         end if;
       end loop;
     end if;
   end if;

   
   
   
   if l_nodename <> '#comment' and l_nodevalue is not null then
     
     
     l_nodevalue := regexp_substr(l_nodevalue, '^\s*(.*?)\s*$', 1, 1, 'n', 1);

     if length(l_nodevalue) > 4000 then
       l_attrvalue := substr(l_nodevalue, 4001);
       l_nodevalue := substr(l_nodevalue, 1, 4000);
       g_message_store_rec.errormessage := substr('Truncated '||to_char(length(l_attrvalue))
                                                              ||' characters from column '
                                                              ||upper(l_nodename)||':'||chr(10)
                                                              ||l_attrvalue||chr(10)
                                                  , 1, 4000);
     end if;

     if l_funname is not null then
       l_nodevalue := execute_function(l_funname, l_nodevalue);
     end if;

     if l_tagdelimit then
       if l_nodevalue is not null then
         
         if l_instruction.instruction = 'RENAME' then
           append_csv_to_parts(l_instruction.newname||'='||l_nodevalue);
         elsif l_instruction.instruction != 'DELETE' then
           append_csv_to_parts(l_nodename||'='||l_nodevalue);
         end if;
       end if;
     else
       
       if g_parts.count > 0 then
         l_previous_path := g_parts(g_parts.count).path;
       else
         l_previous_path := '-';
       end if;
       
       if l_instruction.instruction = 'DENORMALIZE' and not g_sibling_denormalization then
         g_sibling_denormalization := true;
       end if;
       if l_previous_path = l_path and l_instruction.instruction != 'DENORMALIZE' then
         append_csv_to_parts(l_nodevalue);
       else
         l_renamedcolumn := xml2table.column_name(l_nodename, l_colname, l_prefix, false);
         
         
         
         
         if l_renamedcolumn = 'EVENTTIME' then
           
           set_event_timestamp(l_nodevalue, l_denormno, g_config.dateformat, g_config.eventtime_conversion, g_config.use_dbtimezone);
         end if;
         add_to_parts(l_path, l_denormno, upper(l_nodename), l_renamedcolumn, l_nodevalue, l_instruction.instruction);
       end if;
     end if;
   end if;
end;


procedure create_table(p_colnames colnames) is
sqlstm varchar2(8000);
begin
  
  sqlstm := 'CREATE TABLE '||g_target_table_name||chr(10)
  ||'( IDEVENT NUMBER(23,0) NOT NULL'||chr(10)
  ||', KEY NUMBER(23,0) NOT NULL'||chr(10)
  ||', INSERTTS TIMESTAMP NOT NULL'||chr(10)
  ||', EVENTTS TIMESTAMP NOT NULL'||chr(10);
  
  if g_config.store_message_in_table = cmn_constant.g_con_yesno_yes
  then
    sqlstm := sqlstm||', XMLMESSAGE XMLTYPE'||chr(10);
  end if;
  
  for k in 1..p_colnames.count loop
    sqlstm := sqlstm||', '||p_colnames(k)||' VARCHAR2(4000)'||chr(10);
  end loop;
  
  sqlstm  := sqlstm||', CONSTRAINT '||substr(g_target_table_name,1,27)||'_PK PRIMARY KEY(KEY) USING INDEX TABLESPACE INDEX_DATA)';
  
  if g_config.store_message_in_table = cmn_constant.g_con_yesno_yes
  then
     sqlstm  := sqlstm||' XMLTYPE XMLMESSAGE STORE AS BINARY XML ( TABLESPACE LOGGING_SMALL ENABLE STORAGE IN ROW CACHE STORAGE ( INITIAL 8M NEXT 8M ) ) ';
  end if;
  
  sqlstm  := sqlstm||' TABLESPACE USER_DATA';

  if g_tracetag is not null then
    add_to_parts(sqlstm, 0, null, null, null, null);
  end if;

  execute immediate sqlstm;

  sqlstm := 'grant select on '||g_target_table_name||' to public';
  if g_tracetag is not null then
    add_to_parts(sqlstm, 0, null, null, null, null);
  end if;

  execute immediate sqlstm;

  if g_tracetag is not null then
    
    add_to_parts(substr(g_sql_statement,1,4000), 0, null, null, null, null);
  end if;
  
  execute immediate g_sql_statement;
  commit;
  g_message_store_rec.errormessage := 'Created table '||g_target_table_name;
  elaborate_error_message;
end;


procedure add_column_to_table(p_errmsg varchar2) is
msg varchar2(4000) := p_errmsg;
sqlstm varchar2(4000);
colname varchar2(30);
missing_col boolean := true;
added_cols integer := 0;
p1 integer := 0;
p2 integer := 0;
begin
  while (missing_col) and (added_cols < 50) loop 
    begin
      p1 := instr(msg, '"');
      p2 := instr(msg, '"', p1+1);
      colname := substr(msg, p1+1, p2-p1-1);

      sqlstm := 'ALTER TABLE '||g_target_table_name||' ADD ('||colname||' VARCHAR2(4000))';
      if g_tracetag is not null then
        add_to_parts(sqlstm, 0, null, null, null, null);
      end if;
      added_cols := added_cols + 1; 
      execute immediate sqlstm;
      if g_tracetag is not null then
        
        add_to_parts(substr(g_sql_statement,1,4000), 0, null, null, null, null);
      end if;
      
      execute immediate g_sql_statement;
      commit;
      missing_col := false;
    exception when others then
      if (sqlcode = -904) then
        msg := sqlerrm;
        missing_col := true;
      else
        raise;
      end if;
    end;
  end loop;
  g_message_store_rec.errormessage := 'Added '||to_char(added_cols)||' columns to '||g_target_table_name;
  elaborate_error_message;
end;


procedure add_increment(io_colnames in out colnames, io_colvalues in out colvalues, i_colname in varchar2, i_col_value in varchar2) is
l_incremented_colname varchar2(50) := i_colname;
begin
  g_increment := g_increment + 1;
  l_incremented_colname := i_colname||to_char(g_increment);
  io_colnames.extend(1);
  io_colnames(io_colnames.count) := l_incremented_colname;
  io_colvalues.extend(1);
  io_colvalues(io_colvalues.count) := substr(i_col_value,1,4000);
  
end;


procedure extend_cols(io_colnames in out colnames, io_colvalues in out colvalues, i_colname in varchar2, i_colvalue in varchar2) is
l_concatenate boolean := false;
l_new_value varchar2(32767) := trim(i_colvalue);
l_incremented_colname varchar2(50) := i_colname;
begin
  if g_increment > 0 then
    l_incremented_colname := i_colname||to_char(g_increment);
  end if;
  if g_message_store_rec.returncode = 0 then
    if l_new_value is null and g_tracetag is not null then
      
      
      
      l_new_value := ' ';
    end if;
    
    if i_colname is not null and l_new_value is not null then
      begin
        if io_colnames.count > 0 and io_colnames(io_colnames.count) = l_incremented_colname then
          
          l_concatenate := true;
          if trim(l_new_value) is not null then
            l_new_value := io_colvalues(io_colvalues.count)||','||chr(10)||l_new_value;
          end if;
        else
          
          g_increment := 0;
        end if;
        if l_concatenate then
          
          if length(l_new_value) > 4000 then
            io_colvalues(io_colvalues.count) := substr(l_new_value,1,4000);
            add_increment(io_colnames, io_colvalues, i_colname, substr(l_new_value,4001));
          else
            io_colvalues(io_colvalues.count) := l_new_value;
          end if;
        else
          io_colnames.extend(1);
          io_colnames(io_colnames.count) := i_colname;
          io_colvalues.extend(1);
          if length(l_new_value) > 4000 then
            io_colvalues(io_colvalues.count) := substr(l_new_value,1,4000);
            add_increment(io_colnames, io_colvalues, i_colname, substr(l_new_value,4001));
          else
            io_colvalues(io_colvalues.count) := l_new_value;
          end if;
          
        end if;
      exception when others then
        if sqlcode = -6532 then
          
          g_message_store_rec.errormessage := 'Too many columns. Target table is limited to 128 columns.'
          ||chr(10)||'Use DENORMALIZE, TAGDELIMIT or DELETE to reduce the amount of columns.'||chr(10);
          g_duplicate_list := find_dup_columns(io_colnames);
        else
          g_message_store_rec.errormessage := substr(io_colnames(io_colnames.count)||'.'||to_char(length(l_new_value))
          ||chr(10)||sqlerrm||chr(10)||dbms_utility.format_error_backtrace,1,4000);
        end if;
        g_message_store_rec.returncode := -8;
        xml2table.log_message;
      end;
    end if;
  end if;
end;


procedure isolate_root_columns(io_colnames in out colnames, io_colvalues in out colvalues) is
begin
  io_colnames.delete;
  io_colvalues.delete;
  for i in 1..g_parts.count loop
    
    if g_parts(i).denormno = 0 then
      extend_cols(io_colnames, io_colvalues, g_parts(i).colname, g_parts(i).value);
    end if;
  end loop;
end;


procedure save_to_table
(i_colnames         in  colnames
,i_colvalues        in  colvalues
,i_key              in  number
,i_event_timestamp  in  timestamp with time zone
)
is
begin
   if g_message_store_rec.returncode = 0
   then
      begin
         
         
         if g_config.store_message_in_table = cmn_constant.g_con_yesno_yes
         then
            core_table_insert_xml(g_target_table_name
                                 ,i_colnames
                                 ,i_colvalues
                                 ,g_message_store_rec.idevent
                                 ,i_key
                                 ,sys_extract_utc(i_event_timestamp)
                                 );
         else
            core_table_insert(g_target_table_name
                             ,i_colnames
                             ,i_colvalues
                             ,g_message_store_rec.idevent
                             ,i_key
                             ,sys_extract_utc(i_event_timestamp)
                             );
         end if;
      exception
         when others then
            if (sqlcode = -942) 
            then
               create_table(i_colnames);
            elsif (sqlcode = -904) 
            then
               add_column_to_table(sqlerrm);
            else
               if sqlcode = -957
               then
                  
                  g_duplicate_list := find_dup_columns(i_colnames);
               end if;
               raise;
            end if;
      end;
   end if;
end save_to_table;


procedure write_to_core_table is
l_colnames colnames := colnames();
l_colvalues colvalues := colvalues();
l_key number := g_message_store_rec.idevent*999; 
l_event_timestamp timestamp with time zone;
begin
  g_duplicate_list := '';
  if g_denormalize_count > 1 then
    
    for n in 1..g_denormalize_count loop
      
      isolate_root_columns(l_colnames, l_colvalues);
      for i in 1..g_parts.count loop
        if g_parts(i).denormno = n then
          extend_cols(l_colnames, l_colvalues, g_parts(i).colname, g_parts(i).value);
        end if;
      end loop;
      if g_event_timestamps.exists(n) then
        l_event_timestamp := g_event_timestamps(n);
      else
        l_event_timestamp := g_event_timestamp;
      end if;
      save_to_table (l_colnames, l_colvalues, l_key + n + g_key_offset - 1, l_event_timestamp);
    end loop;
  else
    
    for i in 1..g_parts.count loop
      extend_cols(l_colnames, l_colvalues, g_parts(i).colname, g_parts(i).value);
    end loop;
    save_to_table (l_colnames, l_colvalues, l_key + g_key_offset, g_event_timestamp);
  end if;
  g_key_offset := g_key_offset + greatest(1, g_denormalize_count);
end;


procedure process_node(l_node dbms_xmldom.domnode) is
l_ok boolean;
l_nodename varchar2(50);
begin
  g_denormalize_count := 0;
  g_event_timestamps.delete;
  g_sibling_denormalization := false;
  begin
    l_nodename := dbms_xmldom.getnodename(l_node);
    if g_message_store_rec.rootnode is null then
      g_message_store_rec.rootnode := l_nodename;
    else
      g_message_store_rec.rootnode := g_message_store_rec.rootnode||','||l_nodename;
    end if;
    g_config := null;
    begin
      g_config := g_configurations(l_nodename);
      g_target_table_name := upper(substr(g_config.tableprefix||l_nodename,1,30));
    exception when others then
      
      g_config := null;
      g_target_table_name := null;
    end;

    if g_target_table_name is null then
      g_message_store_rec.errormessage := l_nodename||' is not configured for processing.';
      g_message_store_rec.returncode := -2; 
      xml2table.log_message;
    else
      extract_node(l_node, null, null, false, 0);
      elaborate_error_message;
    end if;

  exception when others then
    g_message_store_rec.errormessage := substr(sqlerrm||chr(10)||dbms_utility.format_error_backtrace,1,4000);
    g_message_store_rec.returncode := -7; 
    xml2table.log_message;
    if g_tracetag is not null then xml2table.dump_parts_to_trace_table; end if;
  end;
end;

procedure mark_siblings_for_denorm is
l_denorm_path varchar2(4000) := '-';
l_len integer := 0;
l_denormno number := 0;
begin
  for i in 1..g_parts.count loop
    if g_parts(i).instruction = 'DENORMALIZE' and g_parts(i).tag is not null then
      l_denormno := g_parts(i).denormno;
      l_len := length(g_parts(i).path) - length(g_parts(i).tag);
      l_denorm_path := substr(g_parts(i).path, 1, l_len)||'%';
    end if;
    if g_parts(i).path like l_denorm_path and g_parts(i).tag is not null then
      g_parts(i).denormno := l_denormno;
    else
      l_denorm_path := '-';
    end if;
  end loop;
end;













procedure process_xmlmessage(i_xmlmessage in xmltype) is
  l_rootnode            dbms_xmldom.domnode;
  l_root_element        dbms_xmldom.domelement;
  l_rootname            varchar2(100) := '';
  l_rootvalue           xmltype;
  l_ok                  boolean;
  l_xmlmessage          xmltype;
  l_namespace_stripped  varchar2(250) := null;

begin
  g_parts              := parts_array();
  g_split_nodes        := t_nodes();
  g_domdoc             := null;
  g_key_offset         := 0;
  l_namespace_stripped := null;
  
  begin
    if g_tracetag is not null
    then
      delete from  xml2table_tracing where tracetag = g_tracetag;
      commit;
    end if;

    g_domdoc       := dbms_xmldom.newdomdocument(i_xmlmessage);
    l_root_element := dbms_xmldom.getdocumentelement(g_domdoc);
    l_rootnode     := dbms_xmldom.makenode(l_root_element);
    l_rootname     := dbms_xmldom.getnodename(l_rootnode);

    
    
    if instr(l_rootname,':') > 0
    then
       
       dbms_xmldom.freedocument(doc => g_domdoc);
       
       l_namespace_stripped := substr(l_rootname,1,instr(l_rootname,':'));
       l_xmlmessage         := xmltype(replace(i_xmlmessage.getclobval(),l_namespace_stripped,null));
       
       g_domdoc       := dbms_xmldom.newdomdocument(l_xmlmessage);
       l_root_element := dbms_xmldom.getdocumentelement(g_domdoc);
       l_rootnode     := dbms_xmldom.makenode(l_root_element);
       l_rootname     := dbms_xmldom.getnodename(l_rootnode);
    end if;

    g_phase := 1;
    xml2table.process_node(l_rootnode);

    if g_message_store_rec.returncode = 0 then
      g_phase := 2;
      if g_sibling_denormalization then
        mark_siblings_for_denorm;
      end if;
      xml2table.write_to_core_table;
    end if;
    if g_tracetag is not null then xml2table.dump_parts_to_trace_table; end if;






    
    for i in 1..g_split_nodes.count loop
      exit when g_message_store_rec.returncode <> 0;
      g_parts.delete;
      g_phase := 3;
      l_rootname := g_message_store_rec.rootnode;
      xml2table.process_node(g_split_nodes(i));
      g_message_store_rec.rootnode := l_rootname;

      if g_message_store_rec.returncode = 0 then
        g_phase := 4;
        if g_sibling_denormalization then
          mark_siblings_for_denorm;
        end if;
        xml2table.write_to_core_table;
      end if;
      if g_tracetag is not null then xml2table.dump_parts_to_trace_table(i); end if;

    end loop;

      if g_config.savemessage = 'Y' and g_message_store_rec.returncode = 0 then
        xml2table.log_message;
      end if;
    
    
    dbms_xmldom.freedocument(doc => g_domdoc);
    g_domdoc := null;
    
  exception when others then
    g_message_store_rec.errormessage := substr(sqlerrm||chr(10)||dbms_utility.format_error_backtrace,1,4000);
    g_message_store_rec.returncode := -9; 
    xml2table.log_message;
    if g_tracetag is not null then xml2table.dump_parts_to_trace_table; end if;

    
    dbms_xmldom.freedocument(doc => g_domdoc);
    g_domdoc := null;
  end;
end;


procedure init_log_message is
begin
  g_message_store_rec.idevent := xml2table_idevent_sequence.nextval;
  g_message_store_rec.insertts := sys_extract_utc(systimestamp);
  g_message_store_rec.eventts := g_message_store_rec.insertts;
  g_message_store_rec.rootnode := null;
  g_message_store_rec.returncode := 0;
  g_message_store_rec.errormessage := null;
  g_message_store_rec.duration := 0;
  g_phase := 0;
end;

  xml2table.process_message(i_message, null);
end;

begin
  xml2table.process_message(i_message, null);
end;

procedure process_message(i_message in clob, i_tracetag in varchar2) is


l_xmlmessage xmltype;
l_reset_active varchar2(1) := 'N';
begin

  l_reset_active    := nvl(cmn_common.get_parameter_value('ResetActive'),'N');
  if l_reset_active = 'Y'
  then
    
    return;
  end if;
  
  init_log_message;
  g_tracetag := i_tracetag;
  begin
    l_xmlmessage := xmltype(i_message);
    g_message_store_rec.message := l_xmlmessage;
  exception when others then
    g_message_store_rec.errormessage := substr(sqlerrm||chr(10)||dbms_utility.format_error_backtrace,1,4000);
    g_message_store_rec.returncode := -1; 
    begin
      l_xmlmessage := xmltype('<Bad><!-- '||replace(replace(i_message,'>','}'),'<','{')||' --></Bad>');
    exception when others then
      l_xmlmessage := null; 
    end;
    g_message_store_rec.message := l_xmlmessage;
    xml2table.log_message;
    return;
  end;
  xml2table.process_xmlmessage(l_xmlmessage);
end;

procedure configure
is
   l_config       configuration_type;
   l_instruction  instruction_type;
begin
   g_configurations.delete;
   
   for c_rec in (select rootnode
                 ,      tableprefix
                 ,      savemessage
                 ,      dateformat
                 ,      eventtime_conversion
                 ,      use_dbtimezone
                 ,      store_message_in_table
                 from   xml2table_configuration
                )
   loop
     l_config.tableprefix            := c_rec.tableprefix;
     l_config.savemessage            := c_rec.savemessage;
     l_config.dateformat             := c_rec.dateformat;
     l_config.eventtime_conversion   := c_rec.eventtime_conversion;
     l_config.use_dbtimezone         := c_rec.use_dbtimezone;
     l_config.store_message_in_table := c_rec.store_message_in_table;
     g_configurations(c_rec.rootnode) := l_config;
   end loop;
   
   g_path_instructions.delete;
    
   for c_rec in (select path
                 ,      instruction
                 ,      newname
                 from   xml2table_instruction
                )
   loop
      l_instruction.instruction := c_rec.instruction;
      if c_rec.instruction in ('TAGDELIMIT','FIRSTCHILDNAME')
      then
         l_instruction.newname := xml2table.valid_oracle_name(c_rec.newname);
      else
         l_instruction.newname := c_rec.newname;
      end if;
      g_path_instructions(c_rec.path) := l_instruction;
   end loop;
end;


begin
  xml2table.load_reserved_words;
  xml2table.configure;
end xml2table;
